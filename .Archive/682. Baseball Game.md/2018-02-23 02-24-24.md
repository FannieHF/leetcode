def calPoints(self, ops):
    sum = 0
    valid = []
    for op in ops:
        if op == "+":
            valid.append(valid[-2]+valid[-1])
            sum += valid[-1]
        elif op == "C":
            sum -= valid.pop()
        elif op == "D":
            valid.append(valid[-1]*2)
            sum += valid[-1]
        else:
            valid.append(int(op))
            sum += valid[-1]
    return sum
