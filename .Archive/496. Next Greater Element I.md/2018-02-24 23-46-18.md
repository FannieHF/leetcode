```python
def nextGreaterElement(self, findNums, nums):
    output = []
    for num in findNums:
        i = nums.index(num)+1
        nextGreater = -1
        for compare in nums[i:]:
            if compare>num:
                nextGreater = compare
                break
        output.append(nextGreater)
    return output
```

```ptyhon
def nextGreaterElement(self, findNums, nums):
    st, d = [], {}

    for v in nums:
        while len(st) and st[-1] < v:
            d[st.pop()] = v
        st.append(v)

    return map(lambda x: d.get(x, -1), findNums)
```
