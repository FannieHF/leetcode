```python
def selfDividingNumbers(self, left, right):
    output = []
    for num in range(left,right+1):
        if self.selfDividable(num):
            output.append(num)
    return output
    
def selfDividable(self,num):
    for i in str(num):
        if int(i)==0 or num%int(i) != 0:
            return False
    return True
```
