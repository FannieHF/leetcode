def isToeplitzMatrix(self, matrix):
        """
        :type matrix: List[List[int]]
        :rtype: bool
        """
        
        for i in range(0, len(matrix)):
            for j in range(0, len(matrix[0])):
                if matrix[i][j] == None:
                    break
                for n in range(1,min(len(matrix[0])-j,len(matrix)-i)):
                    if matrix[i+n][j+n] != matrix[i][j]:
                        return False
                    matrix[i+n][j+n] = None
                matrix[i][j] = None
        return True
