```python
def mergeTrees(self, t1, t2):
    t1 = self.mergeNode(t1,t2)
    return t1

def mergeNode(self,t1,t2):
    if not t1 :
        return t2
    elif t2 !=None and t1 != None:
        t1.val =  t1.val+t2.val
        t1.left = self.mergeNode(t1.left,t2.left)
        t1.right = self.mergeNode(t1.right, t2.right)
    return t1
```
