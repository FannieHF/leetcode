```python3
class Solution:
    def findBottomLeftValue(self, root):
        level_list = collections.defaultdict(list)
        
        def bfs(root,level):
            if not root:
                return
            bfs(root.left,level+1)
            level_list[level].append(root.val)
            bfs(root.right,level+1)
        
        bfs(root,0)
        return level_list[len(level_list)-1][0]
```

##  bfs
```python3
class Solution:
    def findBottomLeftValue(self, root):
        pre = [root]
        while pre:
            line = pre
            pre = self.bfs(pre)
        return line[0].val
    
    def bfs(self,pre):
        if not pre:
            return pre
        level = []
        for node in pre:
            if node.left:
                level.append(node.left)
            if node.right:
                level.append(node.right)
        return level
```

## briliant BFS right-to-left
```python3
class Solution:
    def findBottomLeftValue(self, root):
        
        line = [root]
        while line:
            node = line.pop(0)
            last = node
            if node.right:
                line.append(node.right)
            if node.left:
                line.append(node.left)
        return last.val
```
