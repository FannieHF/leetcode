def fizzBuzz(self, n):
    output = []
    for i in range(1,n+1):
        if not i%15:
            output.append("FizzBuzz")
        elif not i%3:
            output.append("Fizz")
        elif not i%5:
            output.append("Buzz")
        else:
            output.append(str(i))
    return output
