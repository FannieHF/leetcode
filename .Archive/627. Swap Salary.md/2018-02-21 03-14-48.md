##1. case
```sql
UPDATE salary SET sex = CASE
    WHEN sex = "f" THEN "m"
    WHEN sex = "m" THEN "f"
    END
```

##2. **^**
```sql
update salary set sex = 
    CHAR(ASCII('f') ^ ASCII('m') ^ ASCII(sex));

# XOR
# 如果sex = "m"， 回合
```

##3. if
```sql
UPDATE salary SET sex = IF(sex='m','f','m');
```
