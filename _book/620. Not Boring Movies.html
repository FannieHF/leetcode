620. Not Boring Movies

1. ! + not

SELECT * FROM cinema WHERE
id % 2 != 0 AND not description = "boring"
order by rating DESC

2. mod + <>

SELECT * FROM cinema WHERE
mod(id,2) = 1 AND description <> "boring"
order by rating DESC
